#------------------------------------------
#
# Simple blog application
#
# This is a simple blog application written 
# to demonstrate the use of Ruby objects.  
# It will allow the user to create and name  
# a blog, write entries to that blog, and 
# print the blog to the screen.  The 
# application only allows one paragraph per 
# blog entry, and does not store the entries 
# to a file or database.
#
# Blog entries are displayed in 
# chronological order.
#
#------------------------------------------


class Blog
attr_accessor :title, :date_created, :blog_author
	
	#------------------------------------------
	# Initialize blog
	#------------------------------------------

	def initialize
		puts "Let's create a new blog!  Enter a name for your blog: "
		@title = gets.chomp
		puts "Enter your name as the creator of this blog: "
		@blog_author = gets.chomp
		@date_created = timestamp
		@blog_hash = {}
		@num_entries = 0
	end
	
	#------------------------------------------
	# Create a new blog entry
	#------------------------------------------

	def entry_create
		#puts "In entry_create function"
		new_timestamp = timestamp
		new_entry = Blog_entry.new(new_timestamp)
		@blog_hash.store(new_timestamp, new_entry)
		@num_entries += 1
	end
		
	#------------------------------------------
	# Print the blog entries in chronological
	# order
	#------------------------------------------

	def blog_print
		print "\n"
		puts "There are currently #{@num_entries} entries in the blog."
		@blog_hash.each{|key, value| 
			25.times do |i|
				print "_"
			end
			print "\nOn ", key, " "
			b_entry = value
			print b_entry.author, " wrote: \n\n"
			print b_entry.title.upcase, "\n\n"
			print b_entry.content, "\n\n\n"
		}
	end
	
end

class Blog_entry 
attr_accessor :title, :author, :datetime, :content

	#------------------------------------------
	# Initialize a blog entry
	#------------------------------------------

	def initialize(time_stamp)
		puts "\nPlease enter a title for your blog post:"
		@title = gets.chomp.capitalize
		puts "\nPlease enter your name as the author of the post:"
		@author = gets.chomp
		puts "\nEnter the content for the blog post:"
		@content = gets.chomp
		@createdate = time_stamp
	end
	
end	

#------------------------------------------
# function timestamp
#
# Purpose: Converts times to GMT to keep 
# entries from different users in different
# time zones in chronological order
#------------------------------------------
def timestamp
	t = Time.new
	t.gmtime
	return t
end

#------------------------------------------
# Main
#------------------------------------------

quit = false
blog = Blog.new
while (! quit)	
	puts "\nWould you like to create a new blog entry (Y/N/Q)?"		
	answer = gets.chomp.upcase
	if (answer == 'Y')
		blog.entry_create
	elsif (answer == 'N')
		puts "\nWould you like to view the entire blog (Y/N/Q)?"
		answer = gets.chomp.upcase
		if (answer == 'Y')
			puts "\n\n#{blog.title}, created on #{blog.date_created} by #{blog.blog_author}.\n\n"
			blog.blog_print
		elsif (answer == 'Q')
			quit = true
		else
			puts "\nThose are the only available actions. Type 'Y' to see them again, or 'N' or 'Q' to quit.(Y/N/Q)?"
			answer = gets.chomp.upcase
			if (answer != "Y")
				quit = true
			end
		end
	elsif (answer == 'Q')
		quit = true
	end
end
